public class Subtract2Numbers {
    public static void main(String[] args) {
        // Define two numbers
        int number1 = 10;
        int number2 = 5;

        // Subtract the numbers
        int difference = number1 - number2;

        // Print the result
        System.out.println("The difference of " + number1 + " and " + number2 + " is: " + difference);
    }
}
